Mario Carneiro (Oct 18 2020 at 14:04):

lemma pnat.strong_induction_on {p : pnat → Prop} : ∀ (n : pnat) (h : ∀ k, (∀ m, m < k → p m) → p k), p n
| n := λ IH, IH _ (λ a h, pnat.strong_induction_on a IH)
using_well_founded {
  rel_tac := λ _ _, `[exact ⟨_, measure_wf coe⟩],
  dec_tac := `[assumption] }
