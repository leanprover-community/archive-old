Mario Carneiro (Aug 14 2020 at 22:28):

example {x : ℝ} : (∀ n : ℕ, x < (n + 1)⁻¹) → x ≤ 0 :=
begin
  contrapose!,
  intro h,
  obtain ⟨n, h⟩ := exists_nat_one_div_lt h,
  use n,
  simpa using le_of_lt h
end
