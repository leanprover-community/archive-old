 (assume (x : X) (hh₁ : (g ∘ f) x = z),

   --exists.intro (f x) hh₂
   sorry),
 h₂

theorem challenge4_tactic
 (X Y Z : Type)
 (f : X → Y)
 (g : Y → Z) :
surjective (g ∘ f) → surjective g :=
begin
 intro h,
 intro z,
 cases h z with a ha,
 use f a,
 assumption,
end
