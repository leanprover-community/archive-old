Yakov Pechersky (Dec 02 2020 at 21:50):

import tactic.basic

def f (x:ℕ) := if h:x=0 then 0 else 1

theorem my_thm (x:ℕ) : if h:x=0 then f x = 0 else f x = 1 :=
begin
  rw f,
  by_cases h : (x = 0),
  { simp [h] },
  { simp [h] },
end

Yakov Pechersky (Dec 02 2020 at 21:51):

Or

import tactic.basic

def f (x:ℕ) := if h:x=0 then 0 else 1

theorem my_thm (x:ℕ) : if h:x=0 then f x = 0 else f x = 1 :=
begin
  split_ifs with h,
  { simp [h, f] },
  { simp [h, f] },
end
