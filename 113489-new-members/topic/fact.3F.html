For the record, this works:

lemma obvious (p : ℕ) [h : fact (nat.prime p)] :
  nat.prime p :=
fact.out _

and also 

lemma obvious (p : ℕ) [h : fact (nat.prime p)] :
  nat.prime p :=
fact.elim h
