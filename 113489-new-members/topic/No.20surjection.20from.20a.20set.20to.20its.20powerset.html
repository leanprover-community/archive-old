Kevin Buzzard (Oct 26 2020 at 15:33):

When I do this with Imperial 1st years I just use types: 

import data.set

open_locale classical

open set

example (S : Type) (f : S ‚Üí set S) : ¬¨ function.surjective f :=
begin
  intro h,
  cases h { b : S | b ‚àâ f b} with x hx,
  by_cases k : x ‚àà f x,
  { have : x ‚àâ f x,
    { rw hx at k, exact k },
    contradiction, },
  { have : x ‚àà f x,
    { rwa hx, },
    contradiction, },
end
