You are indeed correct: Lean thinks you want the power to remain in the submodule. You can use a type ascription to hint the expected type:

import data.mv_polynomial.basic
import ring_theory.polynomial.homogeneous

open_locale big_operators

def is_waring_decomposition {σ : Type*} {R : Type*} [comm_semiring R]
  (p : mv_polynomial σ R) (s : finset (ℕ × mv_polynomial.homogeneous_submodule σ R 1)) :=
(∑ i in s, (i.2 : mv_polynomial σ R)^i.1) = p
