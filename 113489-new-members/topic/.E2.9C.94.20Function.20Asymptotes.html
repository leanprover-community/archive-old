To finish off the thread I tried to generalize this into a proposition which is true when a function from R to R has a vertical/horizontal asymptote:

def vertical_asymptote
    (a : ℝ)
    (f : ℝ → ℝ) :=
    filter.tendsto f (𝓝[>] a) filter.at_top ∨ -- limit as x → a⁺ of f is ∞
    filter.tendsto f (𝓝[>] a) filter.at_bot ∨ -- limit as x → a⁺ of f is -∞
    filter.tendsto f (𝓝[<] a) filter.at_top ∨ -- limit as x → a⁻ of f is ∞
    filter.tendsto f (𝓝[<] a) filter.at_bot -- limit as x → a⁻ of f is -∞

def horizontal_asymptote
    (a : ℝ)
    (f : ℝ → ℝ) :=
    filter.tendsto f filter.at_top (𝓝 a) ∨ -- limit as x → ∞ of f is a
    filter.tendsto f filter.at_bot (𝓝 a) -- limit as x → -∞ of f is a

I came up with this: 

def f_asy
    (a f : ℝ → ℝ) :=
    -- as x gets arbitrarily large the fuctions get arbitrarily close
    filter.tendsto (λ x, | f x - a x |) filter.at_top (𝓝 0)  ∧
    -- as x gets arbitrarily small the fuctions get arbitrarily close
    filter.tendsto (λ x, | f x - a x |) filter.at_bot (𝓝 0)
