Alex J. Best (Feb 21 2022 at 15:08):

To answer the question about getting the nnreal version from the real, you can use the mod_cast tactics

import data.real.nnreal
theorem full'
  (a b : real)
  (h₁ : (a * b)=180)
  (h₂ : 2 * (a + b)=54) :
  (a^2 + b^2) = 369 :=
  sorry
theorem full
  (a b : nnreal)
  (h₁ : (a * b)=180)
  (h₂ : 2 * (a + b)=54) :
  (a^2 + b^2) = 369 :=
begin
  exact_mod_cast (full' a b _ _),
  assumption_mod_cast,
  assumption_mod_cast,
end
