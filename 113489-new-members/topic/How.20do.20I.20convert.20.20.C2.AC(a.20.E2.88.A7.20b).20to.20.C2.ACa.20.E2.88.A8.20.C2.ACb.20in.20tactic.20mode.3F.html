Here is a constructive term mode proof.

variables (p q r : Prop)

theorem implies_commutative : ((p → q) → r) → (p → (q → r)) :=

λ hpqr hp hq, hpqr _

  have h4: ¬p,
  apply h2,
  apply h3,
  intro h4,
  have h5 := h2 h4,
  exact h5 h4,
  have h6 := h3 h4,
  exact h4 h6,
end
