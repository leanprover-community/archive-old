Bassem Safieldeen (Dec 07 2020 at 15:21):

lemma sum_nonneg_zero {f : ι → ℝ} (hf1 : ∑ i, f i = 0) (hf2: ∀ i, 0 ≤ f i) :
∀ i, f i = 0 :=
begin
  intro i,
  have H : ∑ i, f i = 0 ↔ ∀ i, f i = 0, {
    rw sum_eq_zero_iff_of_nonneg,
    {
      finish,
    },
    {
      simp only [hf2],
      finish,
    },
  },
  have H2 : ∀ i, f i = 0, {
    exact H.1 hf1,
  },
  apply H2,
end

  finish,
  finish,
end
