/--
[¬P] = 1 - [P]
-/
lemma not_is_one_minus (i j: Iverson)
: (i.p = ¬j.p) → (i.z = 1 - j.z) :=
begin
  intro h₀,
  rw [i.invariant, j.invariant],
  rw [show (ite i.p (1: ℤ) 0) = (ite (¬j.p) 1 0), by congr; exact h₀],
  rw [ite_not, eq_comm],
  exact apply_ite (has_sub.sub 1) j.p 1 0,
end
