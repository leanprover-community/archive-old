here's an induction proof

local attribute [simp]
lemma nat.smul_one (d : ℕ) : d • (1 : ℤ) = (d : ℤ) :=
begin
  induction d with k hk, simp,
  rw nat.succ_eq_add_one, push_cast,
  rw ← hk, rw add_smul, simp,
end
