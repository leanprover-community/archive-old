Kayla Thomas (Nov 20 2022 at 02:57):

Ok, no, I have it now.

example
  {α β : Type}
  (l1 : list α)
  (l2 : list β) :
  list.map prod.fst (l1.zip l2) ⊆ l1 :=
begin
  induction l1 generalizing l2,
  simp,
  induction l2,
  simp,
  simp at *,
  have s1 : l1_tl ⊆ l1_hd :: l1_tl,
  simp,
  transitivity,
  apply l1_ih,
  apply s1,
end
