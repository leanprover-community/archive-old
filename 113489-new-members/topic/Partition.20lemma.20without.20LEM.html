lemma partition_els_eq_if_share_el {X}
  {F : set (set X)}
  (disjoint : ∀ A B ∈ F, A = B ∨ A ∩ B = ∅)
  {A B : set X}
  (h0 : A ∈ F) (h1 : B ∈ F)
  (h2 : ∃ x : X, x ∈ A ∧ x ∈ B) : A = B :=
(disjoint _ _ h0 h1).resolve_right $ λ h,
let ⟨x, hx⟩ := h2 in
show x ∈ (∅ : set X), from h ▸ hx

#print axioms partition_els_eq_if_share_el
-- no axioms
