@Jason Orendorff Try specialize. Funny name, but it's supposed to be like the dual of generalize.

example (b c : Prop) (h : ∀ p : Prop, p → (b ∧ c)) : b ∧ c :=
begin
  intros,
  specialize h true trivial,
  assumption,
end
