Kenny Lau (Jun 27 2019 at 22:52):

import tactic.interactive

def f (n : ℕ) : ℕ :=
match n with
| 0 := 1
| (k+1) := 0
end

example (n:ℕ) (hf : f (n + n) = 2) : true :=
begin
  generalize_hyp h : n + n = m at hf,
  cases m,
  case nat.zero { change 1 = 2 at hf, trivial },
  case nat.succ : m { change 0 = 2 at hf, trivial }
end
