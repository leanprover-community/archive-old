Kenny Lau (Apr 14 2020 at 21:04):

import tactic data.real.basic

lemma z (F : ℕ → ℝ) (G : ℕ → ℝ) (i : ℕ) (fg : ∀ k, G k = F (k+i)) (k : ℕ) :
  F (k + nat.succ i) = G (k + 1) :=
by rw fg; congr' 1; omega
