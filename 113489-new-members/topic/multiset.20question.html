Kenny Lau (Mar 14 2019 at 19:04):

import data.finset

variables (α : Type*) [decidable_eq α]

example (m : multiset α) : m.to_finset = ∅ → m = 0 :=
multiset.induction_on m (λ _, rfl) (λ _ _ _ H, absurd H $ by simp)
