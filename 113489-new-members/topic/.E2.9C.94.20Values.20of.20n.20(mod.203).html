Patrick Johnson (Apr 03 2022 at 14:20):

A more general version (similarly can be implemented for ℤ):

import tactic
import tactic.induction

def prop (n : ℕ) : ℕ → Prop
| 0 := n = 0
| (nat.succ k) := n = k.succ ∨ prop k

lemma prove_le {n k : ℕ} (h : k ≤ n) : prop k n :=
by { induction' h, cases k; simp [prop], exact or.inr ih }

lemma prove_mod (n : ℕ+) {k : ℕ} : prop (k % n.1) (n.1 - 1) :=
begin
  apply prove_le, rcases n with ⟨⟨_|n⟩, h⟩; simp, cases h,
  exact (nat.le_of_lt_succ (nat.mod_lt _ (nat.zero_lt_succ _))),
end

example {n : ℕ} : n % 3 = 2 ∨ n % 3 = 1 ∨ n % 3 = 0 :=
prove_mod 3
