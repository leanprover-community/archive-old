Patrick Johnson (Sep 25 2022 at 04:15):

In set theory it is straightforward:

import set_theory.zfc.basic

example : ¬∃ (a : Set), a.powerset ⊆ a :=
begin
  rintro ⟨a, h⟩,
  refine Set.mem_irrefl _ (@h a _),
  rw Set.mem_powerset,
  refl,
end
