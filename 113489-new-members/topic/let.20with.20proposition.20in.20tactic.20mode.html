import tactic.interactive

open function

meta def tactic.interactive.result : tactic unit :=

theorem cantor_surjective
  (X : Type) (f : X → set X) (Hf : surjective f) : false :=
begin
  let H := Hf {x | x ∉ f x},
  cases H with D e,
  result
  -- ⊢ (let H : ∃ (a : X), f a = {x : X | x ∉ f x} := Hf {x : X | x ∉ f x} in false) (Exists.intro D e)
end
