example (z : ℂ) : z = z.abs • exp (z.arg • I) :=

  obtain rfl|H := eq_or_ne z 0,

  { simp [exp_re, complex.cos_arg H, mul_div_cancel' _ (abs_ne_zero.mpr H)] },
