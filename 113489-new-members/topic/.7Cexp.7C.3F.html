I don't know what PR means? But here's a (very badly, quickly scraped together) proof (each step is testing what's in library_search until I find something that makes progress... Definitely not mathlib ready!...):

theorem absExpExpRe : Π z:ℂ , complex.abs (complex.exp z) = real.exp (complex.re z) :=
begin
    intros,

    have rw1: real.exp (complex.re z) = (complex.exp (complex.re z)).re,
    {
        exact (complex.exp_of_real_re z.re).symm,
    },
    rw rw1,
    have rw2: complex.abs (complex.exp z) = complex.abs (complex.exp ((z.re):ℂ )),
    {
        have rw3: z.re = ((z.re):ℂ ).re,
        {
            simp,
        },
        exact complex.abs_exp_eq_iff_re_eq.mpr rw3,
    },
    rw rw2,

    have rw4 : complex.exp ((z.re):ℂ ) = real.exp (z.re),
    {
        exact (complex.of_real_exp z.re).symm,
    },
    rw rw4,

    have rw5: ∀ y:ℝ , complex.abs (y:ℂ )= abs y,
    {
        exact complex.abs_of_real,
    },
    rw rw5 (real.exp z.re),

    have rw6: ∀ y:ℝ, 0 < real.exp y,
    {
        exact exppos,
    },
    have rw7: ∀ y:ℝ , 0< y → abs y = y,
    {
        intros,
        exact abs_of_pos a,
    },
    refine rw7 (real.exp z.re) (rw6 (z.re)),
end

Yakov Pechersky (Sep 24 2020 at 19:39):

Without changing the order of your proof, some cleanup:

theorem absExpExpRe (z : ℂ) : complex.abs (complex.exp z) = real.exp (complex.re z) :=
begin
  have rw2: complex.abs (complex.exp z) = complex.abs (complex.exp ((z.re) : ℂ)),
  {
    have rw3: z.re = ((z.re):ℂ ).re, by simp,
    exact complex.abs_exp_eq_iff_re_eq.mpr rw3,
  },
  rw ←complex.exp_of_real_re,
  rw rw2,
  rw ←complex.of_real_exp,
  rw complex.abs_of_real,
  rw abs_of_pos (real.exp_pos _),
  refl
end

Ruben Van de Velde (Sep 24 2020 at 19:41):

Or by changing the order quite a bit:

theorem absExpExpRe (z : ℂ) : complex.abs (complex.exp z) = real.exp (complex.re z) :=
calc complex.abs (complex.exp z)
    = complex.abs (complex.exp z.re) : complex.abs_exp_eq_iff_re_eq.mpr (complex.of_real_re z.re)
... = complex.abs (real.exp z.re) : by rw complex.of_real_exp z.re
... = abs (real.exp z.re) : complex.abs_of_real _
... = real.exp (z.re) : abs_of_pos (real.exp_pos _)
