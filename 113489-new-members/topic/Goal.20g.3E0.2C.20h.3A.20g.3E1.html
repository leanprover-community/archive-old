Some options:

import data.real.basic

example (g : ℝ) (h: g > 1) : g > 0 :=
by linarith

example (g : ℝ) (h: g > 1) : g > 0 :=
begin
  transitivity (1 : ℝ),
  exact h,
  norm_num
end

example (g : ℝ) (h: g > 1) : g > 0 :=
calc g > 1 : h
... > 0 : by norm_num
