I was finally able to finish it!!

  theorem and_distrib_or : p ∧ (q ∨ r) ↔ (p ∧ q) ∨ (p ∧ r) :=
    Iff.intro
      (λ h : p ∧ (q ∨ r) =>
        Or.elim (And.right h)
        (λ hq : q =>
          Or.intro_left
            (p ∧ r)
            (And.intro (And.left h) hq))
        (λ hr : r =>
          Or.intro_right
            (p ∧ q)
            (And.intro (And.left h) hr)))
      (λ h : (p ∧ q) ∨ (p ∧ r) =>
        Or.elim h
        (λ hpq : p ∧ q =>
          And.intro
            (And.left hpq)
            (Or.intro_left r (And.right hpq)))
        (λ hpr : p ∧ r =>
          And.intro
            (And.left hpr)
            (Or.intro_right q (And.right hpr))))
