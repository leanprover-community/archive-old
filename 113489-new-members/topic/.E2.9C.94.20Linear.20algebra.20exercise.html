lemma is_isotropic_iff_restrict_eq_zero (B : bilin_form R M) (N : submodule R M) :
  is_isotropic B N ↔ B.restrict N = 0 :=
begin
  rw [bilin_form.ext_iff],
  split,
  { rintro h ⟨x, hx⟩ ⟨y, hy⟩,
    exact h hy x hx },
  { rintro h x hx y hy,
    exact h ⟨y, hy⟩ ⟨x, hx⟩ }
end

example {X : Type*} (s : set X) (x : s) : true :=
begin
  let x' := x.val,
  have : x' ∈ s := x.property,
  -- Better:
  let x'' := (x : X), -- here Lean inserts a coercion map
  have : x'' ∈ s := x.property,
  trivial
end
