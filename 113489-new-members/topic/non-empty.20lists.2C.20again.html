Bryan Gin-ge Chen (Oct 26 2019 at 07:17):

This works:

def llist (V : Type) : Type := V × list V

def simple {V : Type} : llist V → Prop
    | (v,[])   := true
    | (v,w::l) := ¬ v ∈ w::l ∧ simple (w,l)
    using_well_founded {rel_tac := λ _ _, `[exact ⟨_, measure_wf (list.length ∘ prod.snd)⟩]}
