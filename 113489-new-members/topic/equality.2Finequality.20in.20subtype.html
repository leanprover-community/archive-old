Kenny Lau (Dec 29 2020 at 18:03):

import data.set.basic

lemma idiomatic {E : Type*} (s : set E) (a : E) (b : s) (ha : a ∈ s) (h : ↑b ≠ a) : b ≠ ⟨a, ha⟩ :=
subtype.ne_of_val_ne h
