import tactic
import data.complex.basic

example {x y : ℂ}
  (h₁ : ∃ (a b : ℤ), x = a + complex.I * b)
  (h₂ : ∃ (c d : ℤ), y = c + complex.I * d) :
  ∃ (e f : ℤ), x * y = e + complex.I * f :=
begin
  rcases h₁ with ⟨a, b, rfl⟩, rcases h₂ with ⟨c, d, rfl⟩,
  use [a * c - b * d, a * d + b * c], ext; simp,
end
