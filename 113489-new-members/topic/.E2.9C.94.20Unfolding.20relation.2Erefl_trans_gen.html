lemma unpack_transitive_closure {α : Type} {r : α → α → Prop} {x z : α}
  (h : relation.refl_trans_gen r x z) (nontriv : x ≠ z) :
    ∃ y : α, (r x y) ∧ (relation.refl_trans_gen r y z) :=
begin
  cases relation.refl_trans_gen.cases_head h with h' h',
  {
    exfalso,
    exact nontriv h',
  },
  exact h',
end
