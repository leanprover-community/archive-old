import data.fin
import data.matrix.basic

--import algebra.big_operators

variables {A B : matrix (fin 2) (fin 2) ℤ}

open_locale big_operators

lemma lem : (A * B) 1 0 = A 1 0 * B 0 0 + A 1 1 * B 1 0 :=
show (A 1 0 * B 0 0) + (A 1 1 * B 1 0 + 0) = _, by rw add_zero
