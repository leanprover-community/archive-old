variable {α : Type*}

def constant_list : α → ℕ → list α
|z 0 := []
|z (n+1) := z :: (constant_list z n)

theorem ridiculous (z:α) (n:ℕ):
  constant_list z (n+1) = z :: (constant_list z n) := rfl
