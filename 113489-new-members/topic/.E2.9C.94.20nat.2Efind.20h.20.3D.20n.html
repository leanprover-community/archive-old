Patrick Johnson (Dec 13 2021 at 17:48):

Does this lemma already exist? This seems to be the easiest way to prove that nat.find is equal to some given number, but I can't find it in the docs.

import tactic
open_locale classical

example {f : ℕ → Prop} {n : ℕ}
  {h : ∃ (n : ℕ), f n}
  (h₁ : f n)
  (h₂ : ∀ (k : ℕ), k < n → ¬f k) :
  nat.find h = n :=
begin
  apply le_antisymm; by_contra h₃; push_neg at h₃,
  { exact nat.find_min h h₃ h₁ },
  { exact h₂ _ h₃ (nat.find_spec h) },
end
