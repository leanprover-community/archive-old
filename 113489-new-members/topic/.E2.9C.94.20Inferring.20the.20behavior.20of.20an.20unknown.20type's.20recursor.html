parameters {N : Type} {z : N} {s : N → N}
parameters (hz : ∀ x, s x ≠ z) (hs : ∀ x y, s x = s y → x = y)
parameter N.rec : Π {P : N → Sort u}, P z → (Π x, P x → P (s x)) → Π x, P x
variables {P : N → Sort u} {base : P z} {step : Π x, P x → P (s x)}

lemma lem : N.rec base step z = base := sorry

This lem doesn't include the assumptions hz and hs so it's too strong an assumption, but it doesn't matter since hz and hs are satisfied by ℕ, 0, and nat.succ.

Patrick Johnson (Jul 23 2022 at 18:31):

Corrected proof

import tactic

universe u

section

parameters {N : Type} {z : N} {s : N → N}
parameters (hz : ∀ x, s x ≠ z) (hs : ∀ x y, s x = s y → x = y)
parameter N.rec : Π {P : N → Sort u}, P z → (Π x, P x → P (s x)) → Π x, P x
variables {P : N → Sort u} {base : P z} {step : Π x, P x → P (s x)}

include hz hs
lemma lem : N.rec base step z = base := sorry

end

example : false :=
begin
  classical,
  have := @lem ℕ 0 nat.succ nat.succ_ne_zero @nat.succ.inj
    (λ f z s n, dite (f n = Prop) (λ h, eq.mpr h true) (λ _, n.rec_on z s))
    (λ _, Prop) false (λ _ _, false),
  simp * at *,
end
