Your hypothesis hp says "if the fixed number ppp you chose at the beginning divides nnn, then also assume that p2>np^2>np2>n. theorem Equivalence1 (A B : Prop) : (A → B) → (¬ A ∨ B) :=
begin
  intro hAB,
  by_cases A,
  { right,
    apply hAB,
    assumption
  },
  { left,
    assumption
  }
end
