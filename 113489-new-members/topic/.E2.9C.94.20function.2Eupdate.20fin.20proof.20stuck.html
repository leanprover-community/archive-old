Got it.

example
    {α β : Type}
    [decidable_eq α]
    (σ : α → β)
    (n : ℕ)
    (xs : fin n → α)
    (ys : fin n → β)
    (x : α)
    (h1 : ∀ (i : fin n), x ≠ xs i) :
    function.update_fin σ n xs ys x = σ x :=
begin
    induction n,
    case nat.zero
  {
        unfold function.update_fin
    },
  case nat.succ : n ih
  {
        unfold function.update_fin,
        have s1 : x ≠ xs ↑n, apply h1,
        simp only [function.update_noteq s1],
        apply ih,
        intros i,
        by_cases ↑i = ↑n,
        rewrite h, exact s1,
        apply h1
    },
end
