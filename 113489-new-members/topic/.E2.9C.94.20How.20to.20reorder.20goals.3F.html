Patrick Massot (Oct 18 2021 at 20:41):

@Ignat Insarov This tactic is in mathlib but it's definition uses nothing outside of core lib, so you can simply copy it:

open tactic

meta def tactic.interactive.swap (n := 2) : tactic unit :=
do gs ← get_goals,
   match gs.nth (n-1) with
   | (some g) := set_goals (g :: gs.remove_nth (n-1))
   | _        := skip
   end

example : 1 = 1 ∧ 2 = 2 ∧ 3 = 3 ∧ 4 = 4:=
begin
  refine ⟨_, _, _, _⟩,
  swap,
  swap 3,
  all_goals { sorry }
end
