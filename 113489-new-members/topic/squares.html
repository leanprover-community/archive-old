Mario Carneiro (Feb 10 2020 at 15:57):

One option:

import tactic.ring
lemma converse (r s x y z:ℕ) (px: (x:ℤ) = r*r-s*s) (py: y=2*r*s) (pz:z=r*r+s*s) :
 x*x+y*y=z*z :=
begin
  apply int.coe_nat_inj,
  simp [px, py, pz],
  ring,
end
