Kevin Buzzard (Feb 23 2022 at 00:15):

import tactic

example : ∀ (a b : ℤ ) (hb : b ≠ 0), |a * b| ≥ |a| :=
begin
  intros a b,
  intro hb, -- hb : b ≠ 0
  have h1 : |b| ≥ 1,
  -- probably not the quickest way to do it
  { change 1 ≤ |b|, -- mathlib lemmas use ≤ not ≥
    rw le_abs, -- get rid of abs
    rcases lt_trichotomy 0 b with (hb | rfl | hb), -- three cases
    { left, linarith }, -- b < 0
    { contradiction }, -- b = 0 can't happen
    { right, linarith } }, -- b > 0
  sorry,
end

Heather Macbeth (Feb 23 2022 at 00:46):

@Artizuala Here's a way to do this kind of thing with minimal human effort, although requiring extra computational power:

import tactic

example : ∀ (a b : ℤ ) (hb : b ≠ 0), |a * b| ≥ |a| :=
begin
  intros a b hb,
  cases lt_or_gt_of_ne hb;
  cases abs_cases (a * b) with h₁ h₁;
  cases abs_cases a with h₂ h₂;
  nlinarith,
end

example : abs (a * b) = abs a * abs b :=
begin
  rcases lt_trichotomy a 0 with ha | ha | ha,
