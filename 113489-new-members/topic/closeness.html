Shing Tak Lam (Jul 13 2020 at 06:05):

I think linarith might be what you're looking for?

import data.real.basic

notation `|` x `|` := abs x

example (a b c ε ε' : real) :
ε > 0
→ ε' > 0
→ | a - c | ≤ ε
→ | b - c | ≤ ε'
→ | a - b | ≤ ε + ε'  :=
begin
intros e_pos e'_pos hac hbc,
rw abs_le at *,
split;
  linarith
end
