example (A : Prop) : (∀ B : Prop, ((A → B) → A) → A) → (¬¬ A → A) :=
begin
  intros h nnA,
  have choose_B_false := h (false),
  apply choose_B_false,
  intro hnA,
  exact false.elim (nnA hnA),
end
