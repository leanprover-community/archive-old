def eval := ℕ → bool
def g (n : ℕ) (f : eval) (m : ℕ) : bool := if m < n then f m else tt

lemma Lemma2 (n : ℕ) (f : eval) (k : ℕ) (h : k = n) : g n f k = tt :=
have ¬ k < n, from λ h', ne_of_lt h' h,
by simp [g, this]

lemma Lemma3 (n : ℕ) (f : eval) (k : ℕ) (h : k > n) : g n f k = tt :=
have ¬ k < n, from not_lt_of_gt h,
by simp [g, this]
