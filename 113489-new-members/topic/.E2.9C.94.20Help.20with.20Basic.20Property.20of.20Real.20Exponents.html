Kevin Buzzard (Nov 02 2022 at 23:08):

zero_le_one changed too. This is working code:

import tactic
import data.real.nnreal
import analysis.special_functions.pow

open_locale nnreal

lemma real.rpow_eq_one_iff_of_nonneg {x y : ℝ} (hx : 0 ≤ x) :
  x ^ y = 1 ↔ x = 1 ∨ y = 0 :=
begin
  obtain rfl | hy := eq_or_ne y 0,
  { simp },
  split,
  { intro h,
    rw [← real.one_rpow y] at h,
    left,
    exact real.rpow_left_inj_on hy hx zero_le_one h, },
  { rintros (rfl | rfl),
    { exact real.one_rpow _ },
    { exact real.rpow_zero _ } },
end

lemma nnreal.rpow_eq_one_iff {x : ℝ≥0} {y : ℝ} :
  x ^ y = 1 ↔ x = 1 ∨ y = 0 :=
by exact_mod_cast real.rpow_eq_one_iff_of_nonneg x.prop
