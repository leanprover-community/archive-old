from graphviz import Digraph
def show(e):
    g = Digraph(comment='Grundbegriffe')
    g.attr(shape='box')
    for a,b in e:
        g.node(a, shape='box')
        g.node(b, shape='box')
        g.edge(a,b)
    return g
