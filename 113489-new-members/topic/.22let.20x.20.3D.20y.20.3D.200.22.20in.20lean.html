You can't, what you did here is say that for fix x and y your f_additive hypothesis holds which is entirely different from saying that ∀ x y, f (x + y) = f x + f y so you'll have to change your f_additive hpothesis to that. Then if you want to e.g. plug in 0 for x and 1 for y you just do: f_additive 0 1

Patrick Johnson (Apr 16 2022 at 14:47):

There is probably something similar in mathlib already, but I can't find it.

Proof

import tactic

example {f : ℚ → ℚ} {a : ℚ}
  (h : ∀ {a b : ℚ}, f (a + b) = f a + f b) :
  f a = f 1 * a :=
begin
  have hh₁ : ∀ {a b : ℚ}, b ≠ 0 → a * b / b = a,
  { rintro a b h₁, exact (div_eq_iff h₁).mpr rfl },
  have hh₂ : ∀ {a b : ℚ}, b ≠ 0 → a / b * b = a,
  { rintro a b h₁, exact (eq_div_iff h₁).mp rfl },
  have h₁ : ∀ {a b : ℚ}, f (a * b) = f a * b,
  { have h₁ : f 0 = 0,
    { have h₁ := @h 0 0, simp at h₁, exact h₁ },
    have h₂ : ∀ {a : ℚ}, f (-a) = -f a,
    { intro a, have h₂ := @h a (-a), simp at h₂, linarith },
    have h₃ : ∀ {a : ℚ} {n : ℕ}, f (a * n) = f a * n,
    { rintro a n, induction n using nat.strong_induction_on with n ih,
      dsimp at ih, cases n, { simpa },
      { simp [mul_add, h], exact ih _ (nat.lt_succ_self _) }},
    replace h₃ : ∀ {a : ℚ} {n : ℤ}, f (a * n) = f a * n,
    { rintro a n, cases n,
      { simp, exact h₃ },
      { change f (a * -(n + 1 : ℕ)) = f a * -(n + 1 : ℕ),
        simp_rw [mul_neg, h₂, h₃] }},
    have h₄ : ∀ {a : ℚ} {n : ℕ}, 0 < n → f (a / n) = f a / n,
    { rintro a n h₄, specialize @h₃ (a / n) n, have h₅ : (n : ℚ) ≠ 0,
      { contrapose! h₄, rw nat.le_zero_iff, simp at h₄, exact h₄ },
      simp at h₃, rw hh₂ at h₃, symmetry' at h₃,
      replace h₃ : f (a / ↑n) * n / n = f a / n,
      { exact congr_fun (congr_arg has_div.div h₃) n },
      rw hh₁ h₅ at h₃, repeat { assumption }},
    rintro a b, nth_rewrite 0 ←rat.num_div_denom b,
    rw [←mul_div_assoc, h₄ b.pos, h₃, mul_div_assoc, rat.num_div_denom] },
  specialize @h₁ 1 a, simp at h₁, exact h₁,
end

import algebra.module.linear_map

example {f : ℚ → ℚ} {a : ℚ}
  (h : ∀ {a b : ℚ}, f (a + b) = f a + f b) :
  f a = f 1 * a :=
begin
  have h₁ := (add_monoid_hom.mk' _ @h).to_rat_linear_map.map_smul' a 1,
  simp at h₁, rwa mul_comm,
end
