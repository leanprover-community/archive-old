Alex J. Best (Dec 01 2020 at 13:38):

lemma foo {α : Type*} [semiring α] (a b c : α)
  (h : a + b = 0) (h' : a * c = 0) : b * c = 0 :=
begin
   apply_fun (* c) at h,
   simpa [add_mul, h'] using h,
end
