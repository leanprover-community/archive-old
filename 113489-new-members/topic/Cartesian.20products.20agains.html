lemma moo : A ×ˢ (B ∩ C) ⊆ (A ×ˢ B) ∩ (A ×ˢ C) :=
begin
  intro x,
  intro y,
  -- realisation that x and y are not what you think they are
  have h₂ : x ∈ A ×ˢ (B ∩ C) = (x.1 ∈ A ∧ x.2 ∈ B ∩ C), from prod_mk_mem_set_prod_eq,
  -- etc etc
