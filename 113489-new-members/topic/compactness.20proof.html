Kevin Buzzard (Nov 07 2019 at 18:46):

import topology.metric_space.basic

example (X : Type*) [metric_space X] (O : X)
(Hcomp : compact {x : X | dist O x ≤ 1}) (r : ℝ) (hr : r > 0) :
  ∃ (A : finset X), ∀ x : X, ∃ a ∈ A, dist O x ≤ 1 → dist x a ≤ r :=
begin
  sorry
end
