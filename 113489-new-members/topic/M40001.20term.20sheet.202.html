Kevin Buzzard (Dec 09 2019 at 23:26):

import data.real.basic

example (x : ℝ) (hx_w : ℤ)
(hx_h : x = ↑hx_w ∧ x ^ 2 < 3) :
3 > hx_w ^ 2 :=
begin
  cases hx_h with h1 h2,
  rw h1 at h2,
  norm_cast at h2,
  assumption,
end
