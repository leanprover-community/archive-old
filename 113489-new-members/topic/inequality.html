Kevin Buzzard (May 30 2020 at 23:03):

open nat

example {x y : ℕ} {h : x + 1 = y + 2} : x > y :=
begin
  apply nat.lt_of_succ_le,
  rw succ_eq_add_one,
  rw ←nat.add_le_add_iff_le_right 1,
  rw h,
end
