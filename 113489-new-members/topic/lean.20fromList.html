For plus, matching on n too works

def plus (v1 : Vect n) (v2 : Vect n) : Vect  n:=
  match n, v1, v2 with
  | 0, _, _ => Vect.empty
  | (n + 1), Vect.cons a v, Vect.cons b w => Vect.cons (a + b) (plus v w)
