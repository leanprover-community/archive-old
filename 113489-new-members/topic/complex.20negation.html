Ruben Van de Velde (Jan 10 2021 at 13:59):

One option:

import data.complex.basic

example (z_re z_im: ℝ) (h2: z_re = 0) (h: ¬({re := 0, im := z_im} : ℂ) = 0) : 0 < z_re ^ 2 + z_im ^ 2 :=
begin
  rw [h2, zero_pow zero_lt_two, zero_add],
  apply lt_of_le_of_ne (pow_two_nonneg _),
  contrapose! h,
  obtain rfl := pow_eq_zero h.symm,
  refl,
end
