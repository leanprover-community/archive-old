before-rw.png

lemma notin_compl_iff_in (a : set N) (x : N): (x ∉ a.compl) ↔ x ∈ a := begin split, by_cases (a x), intro, exact h, intro b, exfalso, exact b h, intros h1 h2, exact h2 h1, end
