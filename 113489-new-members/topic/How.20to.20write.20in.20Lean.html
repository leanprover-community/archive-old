theorem ZA_AnR: ∀ (a : ℤ), a * 0 = 0 :=
begin
  intro a,

  have h: a + (a * 0) = a,

    calc
    a + (a * 0) = a * 1 + (a * 0) : by rw [int.mul_one a]
            ... = a * (1 + 0)     : by rw ← int.distrib_left
            ... = a * 1           : by rw int.add_zero
            ... = a               : by rw int.mul_one,
  have h' : a + (a * 0) = a + 0, sorry,
