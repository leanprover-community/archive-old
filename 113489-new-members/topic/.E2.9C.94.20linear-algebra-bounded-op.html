lemma bounded_op_eq_one_if_norm_is_one (h : ∀ (ψ : H), ∥ψ∥ = 1 → ⟪ψ, T ψ⟫_ℂ = 1) :
  T = 1 :=
begin
  rw [← sub_eq_zero, ← inner_map_self_eq_zero],
  intro ψ,
  obtain rfl|hψ := eq_or_ne ψ 0,
  { simp },
  obtain ⟨φ, n, hφ, rfl⟩ : ∃ φ (r : ℝ), ∥φ∥ = 1 ∧ ψ = (r : ℂ) • φ,
  { refine ⟨(1 / ∥ψ∥) • ψ, ∥ψ∥, _, _⟩, -- does both `use`s and the `split` in one line
    -- anything but `finish`, I haven't got all day; all found using `squeeze_simp`
    { rw [norm_smul, one_div, norm_inv, norm_norm, inv_mul_cancel (norm_ne_zero_iff.mpr hψ)], },
    { simp only [smul_smul, mul_inv_cancel (norm_ne_zero_iff.mpr hψ), one_div, complex.coe_smul,
        one_smul], }, },
  suffices : ⟪(T - 1) φ, φ⟫_ℂ = 0,
  { rw [inner_smul_right, linear_map.map_smul, inner_smul_left, this],
    -- mathematicians always want to make the intermediate `have` statements. You don't
    -- need them -- just work on the goal directly.
    simp },
  simp only [inner_sub_left, linear_map.sub_apply, linear_map.one_apply], -- non-terminal simps considered harmful
  rw [inner_self_eq_norm_sq_to_K, hφ, ←inner_conj_sym (T φ) φ, h φ hφ],
  norm_num,
end

lemma bounded_op_eq_one_if_norm_is_one' (h : ∀ (ψ : H), ∥ψ∥ = 1 → ⟪ψ, T ψ⟫_ℂ = 1) :
  T = 1 :=
begin
  rw [←sub_eq_zero, ←inner_map_self_eq_zero],
  intro ψ,
  obtain rfl|hψ := eq_or_ne ψ 0,
  { simp only [eq_self_iff_true, inner_zero_right, map_zero, sub_self], },
  obtain ⟨φ, n, hφ, rfl⟩ : ∃ φ (n : ℝ), ∥φ∥ = 1 ∧ ψ = (n : ℂ) • φ,
  { refine ⟨(1 / ∥ψ∥) • ψ, ∥ψ∥, _, _⟩,
    { simp [norm_smul, norm_ne_zero_iff.mpr hψ], },
    { simp [smul_smul, hψ], } },
  suffices : ⟪(T - 1) φ, φ⟫_ℂ = 0,
  { rw [linear_map.map_smul_of_tower, inner_smul_real_left, inner_smul_real_right, this,
      smul_zero, smul_zero] },
  have t2 : ⟪φ,φ⟫_ℂ = 1,
  { rw [inner_self_eq_norm_sq_to_K φ, hφ, complex.of_real_one, one_pow] },
  have hφ' := h φ hφ,
  rw [linear_map.sub_apply, linear_map.one_apply, inner_sub_left, t2, sub_eq_zero,
    ←inner_conj_sym (T φ) φ, hφ', map_one],
end
