Yaël Dillies (May 04 2022 at 20:24):

theorem cantor_surjective' {α : Type} (f : α → set α) :
  ¬function.surjective f :=
begin
  set B := {x : α | x ∉ f x},
  by_contradiction,
  obtain ⟨ξ, fx⟩ : ∃ ξ, f ξ = B := h B,
  have : ξ ∈ B ↔ ξ ∉ f(ξ),
  { exact mem_set_of, },
  { rw fx at this,
    exact p_equiv_np_implies_false _ this }
end
