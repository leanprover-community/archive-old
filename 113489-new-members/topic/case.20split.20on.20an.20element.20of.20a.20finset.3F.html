This works for me though

import tactic
import data.finset
example {α : Type*} {y y1 y2 : α} {h : y ∈ ({y1, y2} : finset α )} : y = y1 ∨ y = y2 :=
begin
  rw finset.mem_insert at h,
  cases h,
  {left, exact h},
  rw finset.mem_singleton at h,
  right
end
