Mario Carneiro (May 26 2021 at 05:18):

import category_theory.functor_category

open category_theory

universes v₁ v₂ u₁ u₂

variables (C : Type u₁) [category.{v₁} C] (D : Type u₂) [category.{v₂} D]

def eval (A : C) : (C ⥤ D) ⥤ D :=
(functor.flip (functor.id _)).obj A
