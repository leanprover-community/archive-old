Amazingly, I managed to find only its twin lemma. What you want is:

import topology.metric_space.basic

open filter
open_locale topological_space filter

/- Move next to `filter.tendsto_prod_iff` -/
lemma filter.tendsto_prod_iff' {Œπ G G'} {f : filter Œπ} {g : filter G} {g' : filter G'}
  {s : Œπ ‚Üí G √ó G'} :
  tendsto s f (g √ó·∂† g') ‚Üî tendsto (Œª n, (s n).1) f g ‚àß tendsto (Œª n, (s n).2) f g' :=
begin
  unfold filter.prod,
  simp only [tendsto_inf, tendsto_comap_iff, iff_self]
end

open filter

lemma prod.tendsto_iff {Œπ G G'} [topological_space G] [topological_space G']
  (seq : Œπ ‚Üí G √ó G') {f : filter Œπ} (x : G √ó G') :
  tendsto seq f (ùìù x)
    ‚Üî tendsto (Œª n, (seq n).fst) f (ùìù x.fst) ‚àß tendsto (Œª n, (seq n).snd) f (ùìù x.snd) :=
begin
  cases x,
  rw [nhds_prod_eq, tendsto_prod_iff'],
end
