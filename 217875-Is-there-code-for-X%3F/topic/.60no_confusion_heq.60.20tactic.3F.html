Patrick Johnson (Nov 11 2022 at 16:59):

Here is a very ugly proof:

example (u : U) (P Q : Σ v, path u v) (h : φ.path_from u P = φ.path_from u Q) :
  φ.path_from u Q = φ.path_from u P :=
begin
  simp only [sigma.mk.inj_iff] at h ⊢, cases h,
  cases P, cases Q, induction P_snd,
  { induction Q_snd,
    { simp },
    { dsimp at *, symmetry' at h_left,
      have h₁ : (φ.obj Q_snd_b ⟶ φ.obj Q_snd_c) = (φ.obj Q_snd_b ⟶ φ.obj u),
      { rw h_left },
      suffices : (path.nil : path (φ.obj u) (φ.obj u)) =
        (φ.map_path Q_snd_ᾰ).cons (h₁.mp (φ.map Q_snd_ᾰ_1)),
      { cases this },
      apply eq_of_heq, apply h_right.trans, congr,
      { exact h_left },
      { symmetry, simp }}},
  { sorry },
end
