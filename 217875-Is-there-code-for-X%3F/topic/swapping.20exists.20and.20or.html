Yakov Pechersky (Oct 07 2020 at 23:11):

example (f g : ℕ → Prop) : ((∃ k : ℕ, f k) ∨ ∃ k : ℕ, g k) ↔ ∃ k : ℕ, f k ∨ g k :=
begin
  split,
  { intro h,
    rcases h with ⟨kf, hf⟩|⟨kg, hg⟩,
    { exact ⟨kf, or.inl hf⟩ },
    { exact ⟨kg, or.inr hg⟩ } },
  { rintro ⟨k, hf | hg⟩,
    { exact or.inl ⟨k, hf⟩ },
    { exact or.inr ⟨k, hg⟩ } },
end

example (f g : ℕ → Prop) : ((∃ k : ℕ, f k) ∨ ∃ k : ℕ, g k) ↔ ∃ k : ℕ, f k ∨ g k :=
by { split; finish }
