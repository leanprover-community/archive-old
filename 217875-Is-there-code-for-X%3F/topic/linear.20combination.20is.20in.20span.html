Jon Eugster (Nov 07 2022 at 18:07):

Thanks for the help, both of you! I ended up using mainly mem_span_iff_total, and with a bit more knowledge of how fin n and finset play together (which I don't have), your suggestions seem very useful, too.

MWE

import algebra.module.submodule.lattice
import data.real.basic
import algebra.module.pi
import data.matrix.notation
import linear_algebra.finsupp
import algebra.big_operators.default

open_locale big_operators
open submodule

variables {R V : Type*} [ring R] [add_comm_monoid V] [module R V]
variables {v : V} {n : ℕ} {b : fin n → V}

lemma mem_span_iff_sum_coeff :
  v ∈ span R (b '' set.univ) ↔ ∃ (coef : fin n →₀ R), v = ∑ i, (coef i) • (b i) :=
begin
  constructor,
  { rw finsupp.mem_span_image_iff_total R,
    intro h,
    rcases h with ⟨coef, h₁, h₂⟩,
    use coef,
    rw ←h₂,
    rw finsupp.total_apply,
    apply finsupp.sum_fintype,
    simp },
  { rw finsupp.mem_span_image_iff_total R,
    intro h,
    cases h with coef h,

    -- put `lift coef to fin n →₀ R using ⟨fintype.of_finite ↥(function.support coef)⟩,`
    -- here to use `(coef : fin n → R)` in the statement.

    use coef,
    constructor,
    { simp },
    { rw h,
      rw finsupp.total_apply,
      apply finsupp.sum_fintype,
      simp }}
end
