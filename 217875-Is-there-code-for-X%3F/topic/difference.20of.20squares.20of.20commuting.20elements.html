lemma commute.mul_self_sub_mul_self_eq {R} [ring R] {a b : R}(h : commute a b) :
  a * a - b * b = (a + b) * (a - b) :=
begin
  rw [add_mul, mul_sub, mul_sub, h.eq],
  abel,
end

lemma commute.mul_self_sub_mul_self_eq' {R} [ring R] {a b : R} (h : commute a b) :
  a * a - b * b = (a - b) * (a + b) :=
begin
  rw [mul_add, sub_mul, sub_mul, h.eq],
  abel,
end
