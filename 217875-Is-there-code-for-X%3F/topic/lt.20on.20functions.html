lemma pi.lt_of_le_of_lt {α : Type*} {β : Type*} [preorder β]
  {f g : α → β} (h : f ≤ g) {x : α} (hx : f x < g x) : f < g :=
begin
  apply lt_of_le_not_le (pi.le_def.mpr h),
  rw [pi.le_def, not_forall],
  refine ⟨x, hx.not_le⟩,
end
