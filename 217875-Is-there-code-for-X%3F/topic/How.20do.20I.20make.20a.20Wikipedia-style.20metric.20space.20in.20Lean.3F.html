  dist_comm := sorry,
  dist_triangle := sorry }

  dist_comm := sorry,
  dist_triangle := sorry }

def t1 : topological_space (ℝ × ℝ) := m1.to_uniform_space.to_topological_space

variables f : ℝ × ℝ → ℝ × ℝ
