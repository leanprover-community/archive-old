Ruben Van de Velde (Jan 23 2023 at 08:29):

I'd use the existing API:

import data.set.intervals.basic

variables {α : Type*} [partial_order α] (a b : α)

namespace set
lemma Icc_subsingleton_of_ge (h : b ≤ a) : set.subsingleton (Icc a b) :=
begin
  obtain rfl|h := h.eq_or_lt,
  { rw [Icc_self],
    exact subsingleton_singleton },
  { rw [Icc_eq_empty h.not_le],
    exact subsingleton_empty }
end
end set
