example {n : ℕ} : (finset.univ : finset {s : finset (fin (1 + n)) // s.card = n}).card = 1 + n :=
begin
  convert @fintype.card_finset_len (fin (1 + n)) _ n,
  simp [add_comm 1 n, nat.add_choose]
end
