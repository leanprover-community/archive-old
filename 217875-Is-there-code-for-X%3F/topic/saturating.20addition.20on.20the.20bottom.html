Yaël Dillies (Jul 30 2021 at 20:51):

Here's my proof of your sorry

import data.polynomial.monic

example (m n : with_bot ℕ) :
  m + n = ⊥ ↔ m = ⊥ ∨ n = ⊥ :=
begin
  split,
  { cases m,
    { exact λ _, or.inl rfl },
    cases n,
    { exact λ _, or.inr rfl },
    exact λ h, (option.some_ne_none _ h).elim },
  { rintro (rfl | rfl);
    simp }
end
