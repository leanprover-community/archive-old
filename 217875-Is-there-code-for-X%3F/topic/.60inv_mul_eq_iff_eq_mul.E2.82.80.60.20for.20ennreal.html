lemma test2 (x y z : ennreal) (hnt : z ≠ ⊤) (hnz : z ≠ 0) : x = z * y ↔ z ⁻¹ * x = y :=
by split; rintro rfl; simp [←mul_assoc, ennreal.inv_mul_cancel, ennreal.mul_inv_cancel, hnt, hnz]
