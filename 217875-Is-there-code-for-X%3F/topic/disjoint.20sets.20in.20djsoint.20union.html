Reid Barton (Nov 24 2020 at 22:09):

example {α β : Type*} (f : α → β) (S T : set α) : function.injective f → S ∩ T = ∅ → (f '' S) ∩ (f '' T) = ∅ := begin
  intros hf hst,
  rw [set.image_inter hf, hst, set.image_empty],
end
