lemma two_le_add_inv {x : ℝ} (hx : 0 < x) : 2 ≤ x + x⁻¹ :=
le_of_mul_le_mul_left
  (sub_nonneg.1 $
    calc 0 ≤ (x - 1) ^ 2 : pow_two_nonneg _
       ... = _ : by field_simp [ne_of_gt hx]; ring)
  hx
