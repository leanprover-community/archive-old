Adam Topaz (Jan 05 2022 at 20:13):

import data.set.finite

example {β : Type*} {n : ℕ} (H : ∀ (s : finset β), s.card ≤ n) :
  (set.univ : set β).finite :=
begin
  haveI : fintype β := fintype_of_finset_card_le n H,
  exact set.finite_univ
end
