import data.nat.cast

lemma semiring.mul_nat_commute {α : Type*} [semiring α] (x : α) (n : ℕ) : x * n = n * x :=
begin
  induction n with n hn,
  { rw [nat.cast_zero, zero_mul, mul_zero] },
  { show x * ↑(n + 1) = ↑(n + 1) * x,
    rw [nat.cast_add, mul_add, hn, nat.cast_one, mul_one],
    rw [← one_mul x, ← mul_assoc, ← add_mul, mul_one, one_mul] }
end
