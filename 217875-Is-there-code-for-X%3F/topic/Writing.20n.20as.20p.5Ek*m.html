This

lemma totient_mul_prime_div {p n : ℕ} (hp : p.prime) (h : p ∣ n) :
  (p * n).totient = p * n.totient :=
begin
  by_cases hzero : n = 0,
  { simp [hzero] },
  { have hfin := (multiplicity.finite_nat_iff.2 ⟨prime.ne_one hp, zero_lt_iff.2 hzero⟩),
    have hpos : 0 < (multiplicity p n).get hfin,
    { refine zero_lt_iff.2 (λ h, _),
      simpa [h] using (multiplicity.is_greatest' hfin (lt_one_iff.mpr h)) },
    obtain ⟨m, hm⟩ := multiplicity.pow_multiplicity_dvd hfin,
    have : ¬ (p ∣ m),
    { rintro ⟨k, hk⟩,
      rw [hk, ← mul_assoc, ← pow_succ'] at hm,
      have h₁ : p ^ ((multiplicity p n).get hfin + 1) ∣ n := ⟨k, hm⟩,
      refine (multiplicity.eq_coe_iff.1 (by simp)).2 h₁ },
    rw [hm, ← mul_assoc, ← pow_succ, mul_comm _ m, mul_comm _ m, nat.totient_mul
      (prime.coprime_pow_of_not_dvd hp this), nat.totient_mul
      (prime.coprime_pow_of_not_dvd hp this), ← mul_assoc p, mul_comm p, mul_assoc m.totient,
      ← succ_pred_eq_of_pos hpos, totient_prime_pow_succ hp, totient_prime_pow_succ hp,
      succ_pred_eq_of_pos hpos, mul_comm _ (p - 1), mul_comm _ (p - 1), ← mul_assoc p,
      mul_comm _ (p - 1), ← mul_assoc m.totient, mul_assoc (p - 1), ← mul_assoc m.totient,
      ← pow_succ, ← succ_pred_eq_of_pos hpos],
    refl }
end

works, but It probably deserves to be splitted.

Ruben Van de Velde (Dec 22 2021 at 18:44):

I managed a proof, but it looks like yours is shorter anyway

example (p n : nat) (hp : p.prime) (hn : 0 < n) : ∃ k m : nat, n = p ^ k * m ∧ ¬(p ∣ m) :=
begin
  have : multiplicity.finite p n,
  { rw multiplicity.finite_nat_iff,
    exact ⟨hp.ne_one, hn⟩ },
  obtain ⟨m, hm⟩ := multiplicity.pow_multiplicity_dvd this,
  refine ⟨(multiplicity p n).get this, m, hm, λ dvd, lt_irrefl (multiplicity p n) _⟩,
  calc multiplicity p n
      < multiplicity p n + multiplicity p m : _
  ... = multiplicity p n : _,
  { rw ←multiplicity.dvd_iff_multiplicity_pos at dvd,
    conv_lhs { rw ←add_zero (multiplicity p n), },
    rwa enat.add_lt_add_iff_left,
    exact multiplicity.ne_top_iff_finite.mpr this },
  { conv_rhs { rw hm, },
    rw hp.multiplicity_mul,
    congr' 1,
    rw multiplicity.multiplicity_pow_self_of_prime,
    simp only [enat.coe_get, eq_self_iff_true],
    rwa ←nat.prime_iff },
end
