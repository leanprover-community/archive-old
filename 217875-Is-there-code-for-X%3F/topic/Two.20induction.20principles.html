def strong_sub_induction {P : ℕ → ℕ → Sort*}
  (H : ∀ n m, (∀ x y, x < n → y < m → P x y) → P n m) : Π (n m : ℕ), P n m
| n m := H n m (λ x y hx hy, strong_sub_induction x y)
