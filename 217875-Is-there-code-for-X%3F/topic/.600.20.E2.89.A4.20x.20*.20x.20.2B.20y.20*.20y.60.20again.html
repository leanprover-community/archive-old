Re: #2637 and the "needs tests" flag -- I posted some tests in the issue, hoping that it was something to work on. I want all of these to be easier:

lemma norm_nonneg (x y : ℝ) : 0 ≤ x * x + y * y :=
begin
  apply add_nonneg;
  apply mul_self_nonneg,
end

lemma norm_eq_zero_iff {x y : ℝ} : x * x + y * y = 0 ↔ x = 0 ∧ y = 0 :=
begin
  split,
  { intro h,
    rw add_eq_zero_iff' at h,
    { simp * at *},
    { apply mul_self_nonneg},
    { apply mul_self_nonneg}},
  { rintros ⟨rfl, rfl⟩,
    simp,
  }
end

lemma norm_nonpos_right {x y : ℝ} (h1 : x * x + y * y ≤ 0) : y = 0 :=
begin
  suffices : x = 0 ∧ y = 0,
    simp [this],
  rw ←norm_eq_zero_iff,
  linarith [norm_nonneg x y],
end

lemma norm_nonpos_left (x y : ℝ) (h1 : x * x + y * y ≤ 0) : x = 0 :=
begin
  rw add_comm at h1,
  apply norm_nonpos_right h1,
end
