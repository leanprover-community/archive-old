Thanks everyone. I've taken @Junyan Xu's proof and generalized it:

lemma cauchy_schwarz {Œ± : Type*} [fintype Œ±] (f g : Œ± ‚Üí ‚Ñù) :
  (‚àë x, f x * g x) ^ 2 ‚â§ (‚àë x, f x ^ 2) * (‚àë x, g x ^ 2) :=
begin
  change euclidean_space ‚Ñù Œ± at f,
  change euclidean_space ‚Ñù Œ± at g,
  have := @abs_inner_le_norm ‚Ñù _ _ _ f g,
  rw [‚Üê abs_norm_eq_norm f, ‚Üê abs_norm_eq_norm g, ‚Üê abs_mul,
    pi_Lp.inner_apply, is_R_or_C.abs_to_real] at this,
  convert sq_le_sq this using 1,
  rw mul_pow,
  iterate 2 { rw [euclidean_space.norm_eq, real.sq_sqrt] },
  { congr; simp only [is_R_or_C.norm_eq_abs, is_R_or_C.abs_to_real, pow_bit0_abs], },
  iterate 2 { exact finset.sum_nonneg' (Œª _, sq_nonneg _) },
end

then created a natural number version and the specialization I needed:

lemma cauchy_schwarz_nat {Œ± : Type*} [fintype Œ±] (f g : Œ± ‚Üí ‚Ñï) :
  (‚àë x, f x * g x) ^ 2 ‚â§ (‚àë x, f x ^ 2) * (‚àë x, g x ^ 2) :=
by exact_mod_cast cauchy_schwarz (Œª x, f x) (Œª x, g x)

  (‚àë x, f x) ^ 2 ‚â§ fintype.card Œ± * ‚àë x, f x ^ 2 :=
by simpa using cauchy_schwarz_nat (Œª _, 1) f
