structure monoid_hom (M : Type*) (N : Type*) [monoid M] [monoid N] :=

structure mul_hom (M : Type*) (N : Type*) [has_mul M] [has_mul N] :=

structure monoid_hom (M : Type*) (N : Type*) [monoid M] [monoid N] extends mul_hom M N :=
