lemma is_open_smul {α : Type*} [normed_group α] [normed_space ℝ α] (X : set α) (hX : is_open X)
  (x : ℝ) (hx : 0 < x) :
  is_open (x • X) :=
begin
  suffices : is_open_map (λ (y:α), x • y),
  { apply this,
    apply hX },
  apply is_open_map_smul',
  apply ne_of_gt hx,
end
