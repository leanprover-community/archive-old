example (n : ℕ) (R : Type u) [semiring R] : (fin n → R) →ₗ[R] (fin (n + 1) → R) :=
{ to_fun := λ v, fin.snoc v 0,
  map_add' := λ x y, by ext t; refine fin.last_cases _ _ t; simp,
  map_smul' := λ x y, by ext t; refine fin.last_cases _ _ t; simp }
