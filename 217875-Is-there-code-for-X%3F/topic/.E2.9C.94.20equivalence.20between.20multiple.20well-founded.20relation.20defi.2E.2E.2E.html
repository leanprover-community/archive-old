I have somewhat nice proofs for no_infinite_descending_chain_of_wf and set_without_min_eq_empty_of_wf:

theorem no_infinite_descending_chain_of_wf (H : well_founded r) :
  ¬ ∃ f : ℕ → α, ∀ n : ℕ, r (f n.succ) (f n) :=
begin
  rintro ⟨f, hf⟩,
  cases H.min_mem _ (set.range_nonempty f) with n hn,
  apply H.not_lt_min,
  { show f _ ∈ set.range f, simp },
  { rw ←hn, exact hf n }
end

theorem set_without_min_eq_empty_of_wf (H : well_founded r) {A : set α} :
  (∀ a ∈ A, ∃ b ∈ A, r b a) → A = ∅ :=
begin
  intro h,
  ext a,
  rw [set.mem_empty_eq, iff_false],
  intro ha,
  rcases h _ (H.min_mem _ _) with ⟨b, hb, hb'⟩,
  exact H.not_lt_min _ (set.nonempty_of_mem ha) hb hb'
end

And here are some kind of ugly proofs for wf_of_no_infinite_descending_chain and wf_of_set_without_min_eq_empty:

noncomputable def not_acc_aux {a : α} (ha : ¬ acc r a) :
  Σ' b, r b a ∧ ¬ acc r b :=
begin
  suffices : ∃ b, r b a ∧ ¬ acc r b,
  { exact ⟨classical.some this, classical.some_spec this⟩ },
  by_contra h,
  push_neg at h,
  exact ha ⟨_, h⟩
end

noncomputable def f {a : α} (ha : ¬ acc r a) : ℕ → Σ' b, ¬ acc r b :=
λ n, nat.rec ⟨a, ha⟩ (λ n a, ⟨(not_acc_aux a.snd).fst, (not_acc_aux a.snd).snd.right⟩) n

theorem wf_of_no_infinite_descending_chain :
  (¬ ∃ f : ℕ → α, ∀ n : ℕ, r (f n.succ) (f n)) → well_founded r :=
begin
  by_contra h,
  simp only [not_forall, exists_prop] at h,
  cases h with h' h, apply h', clear h',
  have : ¬ ∀ a, acc r a := h ∘ well_founded.intro, clear h, rename this h,
  push_neg at h,
  cases h with a ha,
  exact ⟨λ n, (f ha n).fst, λ n, (not_acc_aux (f ha n).snd).snd.left⟩
end

theorem wf_of_set_without_min_eq_empty :
  (∀ A : set α, (∀ a ∈ A, ∃ b ∈ A, r b a) → A = ∅) → well_founded r :=
begin
  by_contra h,
  simp only [exists_prop, not_forall] at h,
  have := mt (@wf_of_no_infinite_descending_chain _ r),
  push_neg at this,
  cases h with h₁ h₂,
  specialize this h₂, clear h₂,
  cases this with f hf,
  let A := set.range f,
  suffices : ∀ a, a ∈ A → ∃ b, b ∈ A ∧ r b a,
  { specialize h₁ A this, finish },
  clear h₁,
  intros a ha,
  have : ∃ n, f n = a,
  { finish },
  rcases this with ⟨n, rfl⟩,
  use f n.succ,
  finish
end
