def function.left_inverse.equiv_range {α β : Sort*} {f' : β → α} {f : α → β} (hf : function.left_inverse f' f):
  α ≃ set.range f :=
{ to_fun := λ a, ⟨f a, a, rfl⟩,
  inv_fun := λ b, f' b,
  left_inv := hf,
  right_inv := λ ⟨b, a, ha⟩, subtype.eq $ show f (f' b) = b,
    from (ha ▸ hf a : f' b = a).symm ▸ ha }
