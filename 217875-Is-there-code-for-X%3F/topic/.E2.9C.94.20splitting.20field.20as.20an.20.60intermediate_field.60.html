Junyan Xu (Aug 01 2022 at 17:56):

Indeed, if we don't insist on showing that the intermediate field is obtained by adjoining the roots, there's a more direct construction using docs#intermediate_field.map, but it requires the image_root_set lemma introduced by one of Thomas's PRs and the fact that is_splitting_field is preserved by alg_equiv:

import analysis.complex.polynomial
import field_theory.adjoin
import field_theory.is_alg_closed.basic

open_locale polynomial
open polynomial

constant p : ℚ[X]

noncomputable def K : intermediate_field ℚ ℂ :=
begin
  let := is_alg_closed.lift (algebra.is_algebraic_of_finite ℚ $ splitting_field p),
  rotate, { exact ℂ }, iterate 4 { apply_instance },
  apply @intermediate_field.map ℚ (splitting_field p),
  exact ⊤, exact this,
end .
/- The following times out :
  intermediate_field.map (is_alg_closed.lift (algebra.is_algebraic_of_finite ℚ $ splitting_field p)) ⊤
  If you don't have the right imports it's also easy to timeout. -/

section

variables {K L L' : Type*} [field K] [field L] [field L'] [algebra K L] [algebra K L']

lemma image_root_set {p : K[X]} (h : p.splits $ algebra_map K L) (f : L →ₐ[K] L') :
  f '' p.root_set L = p.root_set L' :=
begin
  classical,
  rw [root_set, ←finset.coe_image, ←multiset.to_finset_map, ←f.coe_to_ring_hom, ←roots_map ↑f
      ((splits_id_iff_splits (algebra_map K L)).mpr h), map_map, f.comp_algebra_map, ←root_set],
end

lemma polynomial.is_splitting_field.of_alg_equiv (f : L ≃ₐ[K] L') (p : K[X])
  [is_splitting_field K L' p] : is_splitting_field K L p :=
begin
  have : splits (algebra_map K L) p,
  { rw ← f.symm.to_alg_hom.comp_algebra_map,
    exact splits_comp_of_splits _ _ (is_splitting_field.splits L' p) },
  refine ⟨this, subalgebra.map_injective f.to_alg_hom f.injective _⟩,
  rw [algebra.map_top, (algebra.range_top_iff_surjective f.to_alg_hom).2 f.surjective],
  erw [← algebra.adjoin_image, image_root_set this],
  exact is_splitting_field.adjoin_roots L' p,
end

end

instance : is_splitting_field ℚ K p :=
begin
  convert polynomial.is_splitting_field.of_alg_equiv _ p,
  exact splitting_field p, apply_instance, apply_instance, swap, apply_instance,
  change subalgebra.map _ _ ≃ₐ[ℚ] _,
  have : (⊤ : intermediate_field ℚ p.splitting_field).to_subalgebra = ⊤,
  { ext, exact iff.rfl },
  apply (subalgebra.equiv_of_eq _ _ _).trans,
  swap 3, rw this,
  rw algebra.map_top,
  exact (alg_equiv.of_injective_field _).symm,
end
