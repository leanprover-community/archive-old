def quotient_bot_equiv (α : Type*) : quotient (⊥ : setoid α) ≃ α :=
{ to_fun := λ x, quotient.lift_on' x id (λ _ _, id),
  inv_fun := quotient.mk',
  left_inv := λ x,
  begin
    apply quotient.induction_on' x,
    intro x,
    refl,
  end,
  right_inv := λ x, rfl }
