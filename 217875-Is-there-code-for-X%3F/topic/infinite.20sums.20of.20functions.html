lemma pi.has_sum {ι α β : Type*} [add_comm_monoid β] [topological_space β]
  {f : ι → α → β} {g : α → β} : has_sum f g ↔ ∀ x, has_sum (λ i, f i x) (g x) :=
begin
  simp_rw [has_sum, nhds_pi, tendsto_infi, tendsto_comap_iff],
  apply forall_congr,
  intro a,
  convert iff.rfl,
  ext s,
  simp
end
