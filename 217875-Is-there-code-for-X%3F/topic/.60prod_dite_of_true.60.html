Adam Topaz (Aug 25 2021 at 14:19):

Here's a more general proof which can be golfed to around 2 lines if necessary.

import algebra.big_operators.basic

namespace finset

open_locale big_operators

@[to_additive] lemma prod_dite_of_true {α β : Type*} [comm_monoid β]
  {s : finset α} {p : α → Prop} {hp : decidable_pred p}
  (f : Π (x : α), p x → β) (g : Π (x : α), ¬p x → β) (h : ∀ x ∈ s, p x) :
  (∏ x in s, if hx : p x then f x hx else g x hx) =
  ∏ x in s.attach, f x.1 (h x.val x.property) :=
begin
  apply @finset.prod_bij β α {x // x ∈ s} _ s s.attach (λ x, dite (p x) (f x) (g x))
    (λ x, f x (h _ x.2)) (λ x hx, ⟨x,hx⟩) (λ x hx, by simp),
  { intros a ha, dsimp, rw dif_pos },
  { intros a₁ a₂ h₁ h₂ hh, dsimp at hh, apply_fun (λ e, e.1) at hh, exact hh },
  { intros b hb,
    use [b.1, b.2],
    simp },
end

end finset
