Kenny Lau (May 24 2020 at 21:09):

import tactic.basic

example (p q : Prop) : ¬p ∨ ¬q → ¬(p ∧ q) :=
begin
  rintros (hp | hq) ⟨p, q⟩; cc
end

Mario Carneiro (May 24 2020 at 21:28):
