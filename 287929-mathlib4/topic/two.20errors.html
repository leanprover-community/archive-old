Like this

namespace List

def rec' {C : List α → Sort _} (nil : C [])
    (cons : (head : α) → (tail : List α) → C tail → C (head :: tail)) (l : List α) : C l :=
  match l with
  | [] => nil
  | hd::tl => cons hd tl (rec' nil cons tl)

end List
