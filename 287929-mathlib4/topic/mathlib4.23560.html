theorem well_founded.asymmetric {α : Sort*} {r : α → α → Prop} (h : well_founded r) :
  ∀ ⦃a b⦄, r a b → ¬ r b a
| a := λ b hab hba, well_founded.asymmetric hba hab
using_well_founded { rel_tac := λ _ _, `[exact ⟨_, h⟩],
