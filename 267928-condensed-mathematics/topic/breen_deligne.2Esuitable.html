def very_suitable (f : universal_map m n) (r r' : out_param ℝ≥0) (c₁ c₂ : ℝ≥0) : Prop :=
∃ (N k : ℕ) (c' : ℝ≥0), f.bound_by N ∧ f.suitable c₁ c' ∧ r ^ k * N ≤ 1 ∧ c' ≤ r' ^ k * c₂

and

class very_suitable (BD : data) (r r' : out_param ℝ≥0) (c_ : ℕ → ℝ≥0) : Prop :=
