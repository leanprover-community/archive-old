import tactic

example (X : Type) (f : X → ℕ) (h : ¬∀ x, f x = 0) : false :=
begin
  have h1 := h,
  -- h h1: ¬∀ (x : X), f x = 0
  push_neg at h,
  push_neg at h1,
  /-
  h  : ∃ (x : X), f x ≠ 0
  h1 : ∃ (x : X), ¬f x = 0
  -/
  sorry,
end
