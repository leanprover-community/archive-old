mathlib/scripts$ cat nolints_summary.sh
#!/bin/bash

input=nolints.txt

fn=''
cnt=0

while read -r line
do
        case "$line" in
                '-- '*)
                        cnt=0
                        fn=${line:3}
                        ;;
                apply*)
                        cnt=$(($cnt+1))
                        ;;
                '')
                        if [ $cnt -gt 0 ]
                        then
                                echo "$cnt $fn"
                        fi
                        ;;
        esac
done < "$input"

echo $(($cnt+1)) $fn

import io
f = io.open("../scripts/nolints.txt","r",encoding="utf-8")

name = ""
freq = []

for line in f:
    if line[0] == "-":
        name = line[3:-6]
        freq.append((0, name))
    elif freq:
        (count, name) = freq[-1]
        freq[-1] = (count+1, name)

freq.sort()

for count, name in freq:
    if count > 2: break
    print(name, count-1)
