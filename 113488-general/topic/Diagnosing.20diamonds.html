Sebastien Gouezel (Dec 16 2020 at 18:36):

A non-golfed version:

lemma has_scalar.ext {R M : Type*} (a b : has_scalar R M) : a.smul = b.smul → a = b :=
begin
  intro h,
  tactic.unfreeze_local_instances,
  cases a,
  cases b,
  dsimp at h,
  induction h,
  refl,
end
