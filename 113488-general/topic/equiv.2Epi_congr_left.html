def equiv.pi_congr_left {α : Sort u} {β : Sort v} (P : β → Sort w) (e : α ≃ β) :
  (Π a, P (e a)) ≃ (Π b, P b) :=
{ to_fun := λ f x, (e.apply_symm_apply x).rec_on (f (e.symm x)),
  inv_fun := λ f x, f (e x),
