Yaël Dillies (Dec 12 2021 at 20:22):

Here's probably what you meant

import logic.basic
import tactic.ext

variable a : Type*

theorem foo : ∀ (p q : a -> a -> Prop), (∀ x y, p x y -> q x y) → (∀ x y, q x y -> p x y) → p = q :=
begin
  intros p q hpq hqp,
  ext x y,
  exact ⟨hpq x y, hqp x y⟩,
end
