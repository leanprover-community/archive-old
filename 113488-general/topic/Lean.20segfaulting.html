       lst ← mmap infer_type ctx,
       foo ← mfilter is_prop lst,
       g ← mmap revert foo,
       skip

example (b c d e f : Prop) (h₁ : b ∧ c) (h₂ : d → e) (g : Prop) (h₃ : b ∧ f → g) :
  true :=
begin
  -- revert_props, -- causes a segfault
  guard_target (b ∧ f → g) → (d → e) → b ∧ c → true,
  intros,
  trivial
end
