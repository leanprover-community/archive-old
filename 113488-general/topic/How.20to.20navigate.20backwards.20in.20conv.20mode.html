Jiatong Yang (Jul 24 2022 at 13:28):

Conv within conv is a nice option. Thank you very much!

import data.real.basic
example (a b c : ℝ) : (c * b) * a = b * (a * c) :=
begin
  conv {
    to_lhs,
    conv {
      congr,
      rw mul_comm,
    },
    rw mul_assoc,
    congr,
    skip,
    rw mul_comm,
  },
end
