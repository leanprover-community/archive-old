theorem mul_left_cancel (a b c : mynat) (ha : a ≠ 0) : a * b = a * c → b = c :=
begin
induction c with d hd generalizing b,
rw mul_zero,

rw mul_eq_zero_iff at g,
cases g,
exfalso,
exact ha g,
rw g,
refl,

cases b,

exact ha g,
exfalso,
exact succ_ne_zero d g,

repeat {rw add_right_cancel_iff},

case mynat.succ
