class ring (α : Type u)
  extends has_add α
  , has_mul α
  , has_one α
  , has_zero α
  , has_inv α
  :=
    (add_assoc : ∀ a b c : α, a + (b + c) = (a + b) + c)
    (mul_assoc : ∀ a b c : α, a * (b * c) = (a * b) * c)
    (add_left_id : ∀ a : α, 0 + a = a)
    (mul_left_id : ∀ a : α, 1 * a = a)
    (mul_right_id : ∀ a : α, a * 1 = a)
    (add_left_inv : ∀ a : α,  a⁻¹ + a = 0)
    (left_distribute : ∀ a b c : α, a * (b + c) = a * b + a * c)
    (right_distribute : ∀ a b c : α, (a + b) * c = a * c + b * c)

open ring

theorem add_right_inv {A : Type*} [ring A] {a : A} : a + a⁻¹ = 0 :=
begin
  rw ← add_left_id (a + a⁻¹),
  rw ← add_left_inv (a + a⁻¹),
  rw ← add_assoc ((a + a⁻¹)⁻¹) (a + a⁻¹),
  conv in (a + a⁻¹ + (a + a⁻¹))
  begin
    rw ← add_assoc,
    rw add_assoc a⁻¹,
    rw add_left_inv,
    rw add_left_id,
  end,
end
