Eric Wieser (Mar 03 2021 at 16:24):

Definitionally maps_to is much easier to work with than the other version:

lemma set.maps_to_def {α β γ : Type*} (f : α → β) (s : set α) (t : set β) :
  set.maps_to f s t ↔ f '' s ⊆ t :=
begin
  unfold set.maps_to set.image set.subset has_subset.subset set_of,
  -- (∀ ⦃x : α⦄, x ∈ s → f x ∈ t) ↔
  -- (∀ ⦃a : β⦄, (a ∈ λ (b : β), ∃ (a : α), a ∈ s ∧ f a = b) → a ∈ t)
  sorry
end
