@Eric Wieser The following provides some speedup for category_theory.simplicial_object.cech_nerve:

/-- The Čech nerve construction, as a functor from `arrow C`. -/
@[simps]
def cech_nerve : arrow C ⥤ simplicial_object C :=
{ obj := λ f, f.cech_nerve,
  map := λ f g F,
  { app := λ n, wide_pullback.lift
      (wide_pullback.base _ ≫ F.right)
      (λ i, wide_pullback.π _ i ≫ F.left)
      (λ i, by rw [category.assoc, arrow.w, wide_pullback.π_arrow_assoc]),
    naturality' := begin
      intros i j f,
      dsimp,
      ext,
      { rw [category.assoc, wide_pullback.lift_π, wide_pullback.lift_π_assoc,
          category.assoc, wide_pullback.lift_π, wide_pullback.lift_π] },
      { rw [category.assoc, wide_pullback.lift_base, wide_pullback.lift_base_assoc,
          category.assoc, wide_pullback.lift_base, wide_pullback.lift_base] },
    end },
  map_id' := begin
    intros i,
    ext,
    { dsimp, rw [wide_pullback.lift_π, category.id_comp, category.comp_id] },
    { dsimp, rw [wide_pullback.lift_base, category.id_comp, category.comp_id] },
  end,
  map_comp' := begin
    intros f g h F G,
    ext,
    { dsimp,
      rw [wide_pullback.lift_π, category.assoc,
        wide_pullback.lift_π, wide_pullback.lift_π_assoc, category.assoc] },
    { dsimp,
      rw [wide_pullback.lift_base, category.assoc,
        wide_pullback.lift_base, wide_pullback.lift_base_assoc, category.assoc] },
  end }
