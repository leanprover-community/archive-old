Eric Rodriguez (Jun 23 2022 at 11:11):

here's a stronger lemma, but I guess more specific than you'd want as it uses real.sqrt:

import topology.algebra.order.basic
import algebra.order.archimedean
import data.real.sqrt


private lemma exists_rat_sq_btwn_rat' {x y : ℚ} (h : x < y) (hx : 0 < x) :
  ∃ q : ℚ, 0 < q ∧ x < q^2 ∧ q^2 < y :=
begin
  have hy := hx.trans h,
  have key : real.sqrt x < real.sqrt y :=
    (real.sqrt_lt_sqrt_iff $ by exact_mod_cast hx.le).mpr (by exact_mod_cast h),
  obtain ⟨q, hxq, hqy⟩ := exists_rat_btwn key,
  have hq : 0 < (q : ℝ) := (real.sqrt_nonneg x).trans_lt hxq,
  rw [real.sqrt_lt (show 0 ≤ (x : ℝ), by exact_mod_cast hx.le) hq.le] at hxq,
  rw [real.lt_sqrt hq.le] at hqy,
  exact ⟨q, by exact_mod_cast hq, by exact_mod_cast hxq, by exact_mod_cast hqy⟩
end

lemma exists_rat_sq_btwn_rat {x y : ℚ} (h : x < y) (hy : 0 < y) :
  ∃ q : ℚ, 0 < q ∧ x < q^2 ∧ q^2 < y :=
begin
  rcases le_or_lt x 0 with hx|hx,
  { obtain ⟨q, hq, h0q, hqy⟩ := exists_rat_sq_btwn_rat' (half_lt_self hy) (half_pos hy),
    exact ⟨q, hq, by linarith, hqy⟩ },
  { exact exists_rat_sq_btwn_rat' h hx }
end
