Kevin Buzzard (Jan 02 2023 at 21:00):

@[to_additive] -- pos_nonpos_iff
lemma pos_le_one_iff {a : α} : a⁺ ≤ 1 ↔ a ≤ 1 :=
by { rw [m_pos_part_def, sup_le_iff, and_iff_left], exact le_refl _ }
