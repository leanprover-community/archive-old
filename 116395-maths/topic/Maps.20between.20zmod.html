Would it be crazy to do something like this?

import data.zmod.basic

variables (n N : ℕ)

class divides (n N : ℕ) := (cond : n ∣ N)

def proj [divides n N] : zmod N →+* zmod n := zmod.cast_hom (divides.cond : n ∣ N) _
