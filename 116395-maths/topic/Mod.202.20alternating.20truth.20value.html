Patrick Johnson (Jan 13 2023 at 06:44):

import data.nat.parity

example {k : ℕ} : (k.succ % 2 = 0) = ¬(k % 2 = 0) :=
by simp_rw [←nat.even_iff, nat.even_add_one]
