import linear_algebra.matrix

lemma ulm2019_2 (n: ℕ) (A B: matrix (fin n) (fin n) ℝ) (t: fin (n + 1) → ℝ) (Ht: function.injective t):
  (∀ i, (A + t i • B).det = 0)
    ↔ ∃ (V W: subspace ℝ (fin n → ℝ)),
