example (K : Type*) [ring K] [nontrivial K] (k : ℕ) (hk : 2 ≤ k) :
  (X ^ k - X : polynomial K) ≠ 0 :=
begin
  apply ne_zero_of_degree_gt (_ : 1 < degree _),
  rw [sub_eq_add_neg, add_comm, degree_add_eq_of_degree_lt]; simp; exact_mod_cast hk,
end
