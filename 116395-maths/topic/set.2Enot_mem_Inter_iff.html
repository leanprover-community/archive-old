Joe (Jan 13 2020 at 22:21):

import data.set.lattice

noncomputable theory
open_locale classical

open set

def set.not_mem_Inter {X : Type*} {ι : Type*} (f : ι → set X) (x : X) :
(x ∉ ⋂ i, f i) ↔ ∃ i : ι, x ∉ f i := by rw [mem_Inter, not_forall]
