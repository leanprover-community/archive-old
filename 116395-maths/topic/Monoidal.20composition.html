example (A W X Y Z : C) (f : W ⟶ X) (g : Y ⟶ Z) [monoidal_coherence X Y] :
  (𝟙 A) ⊗ (f ⊗≫ g) = ((𝟙 A) ⊗ f) ⊗≫ ((𝟙 A) ⊗ g) :=
by simp [tensor_hom, monoidal_comp]
