Kevin Buzzard (Dec 30 2019 at 09:47):

opens.Sup_s is helpful!

lemma what_is_this_called {X Y : Type*} [topological_space X] [topological_space Y] {f : X → Y}
  (hf : continuous f) {γ : Type*}
  (ι : γ → opens Y) :
(⨆ (j : γ), hf.comap (ι j)).val = ⋃ (j : γ), f ⁻¹' (ι j).val :=
begin
  unfold lattice.supr,
  rw opens.Sup_s,
  ext x,
  split,
    rintro ⟨V, HV, H⟩,
    use V,
    existsi _,
      exact H,
    convert HV,
    ext Z,
    split,
      rintro ⟨i, rfl⟩,
      split,
        split,
          use i,
      refl,
    rintro ⟨_, ⟨i, rfl⟩, rfl⟩,
    use i,
    refl,
  rintro ⟨_, ⟨j, rfl⟩, h3⟩,
  rw set.mem_sUnion,
  use f ⁻¹' (ι j).val,
  existsi _,
    exact h3,
  existsi hf.comap (ι j),
  split,
    use j,
  refl,
end
