Abdalrhman M Mohamed (Sep 05 2021 at 16:51):

For the record, I managed to resolve the issue by adding 2 lemmas:

theorem ite_then {x y : α} (hp : p) [s : Decidable p] : (if p then x else y) = x :=
  match s with
  | isTrue hp => rfl
  | isFalse hnp => absurd hp hnp

theorem ite_else {x y : α} (hnp : ¬p) [s : Decidable p] : (if p then x else y) = y :=
  match s with
  | isTrue hp => absurd hp hnp
  | isFalse hnp => rfl

theorem apply_update' [DecidableEq α] (m : Map α β) : m[k ↦ v] k = v :=
  show (if k = k then v else m k) = v from
  ite_then rfl
