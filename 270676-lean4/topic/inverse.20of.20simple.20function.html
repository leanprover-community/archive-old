inductive Tile where
| empty
| wall
| player
| target
| box
deriving Inhabited


def Tile.toChar : Tile -> Char
| .wall => '#'
| .empty => ' '
| .target => '.'
| .player => '@'
| .box => 'o'

def Tile.parseP? : (c: Char) -> Option {t: Tile // c = t.toChar}
| '#' => .some ⟨.wall,   (by simp)⟩
| ' ' => .some ⟨.empty,  (by simp)⟩
| '.' => .some ⟨.target, (by simp)⟩
| '@' => .some ⟨.player, (by simp)⟩
| 'o' => .some ⟨.box,    (by simp)⟩
| other => .none
