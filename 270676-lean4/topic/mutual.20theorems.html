Reid Barton (Jun 20 2022 at 12:14):

Lean 3 example

mutual theorem a, b
  with a : ∀ (n : ℕ), n = n
  | 0 := rfl
  | (n + 1) := @congr_arg _ _ _ _ (λ i, i + 1) (b n)
  with b : ∀ (n : ℕ), n = n
  | 0 := rfl
  | (n + 1) := @congr_arg _ _ _ _ (λ i, i + 1) (a n)

#print a        --  theorem a : ∀ (n : ℕ), n = n := a._main
#print a._main  --  theorem a._main : ∀ (n : ℕ), n = n := λ (n : ℕ), a.b._mutual (psum.inl n)
