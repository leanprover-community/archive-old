Ruben's suggestion also works:

theorem bodd_succ (n : ℕ) : bodd (succ n) = not (bodd n) := by
  unfold bodd
  conv =>
    lhs
    unfold boddDiv2
  let ⟨b,m⟩ := boddDiv2 n
  cases b <;> rfl

Kevin Buzzard (Dec 19 2022 at 16:39):

change works too:

theorem bodd_succ (n : ℕ) : bodd (succ n) = not (bodd n) := by
  unfold bodd
  change (match boddDiv2 n with
    | (false, m) => (true, m)
    | (true, m) => (false, succ m)).fst =
  !(boddDiv2 n).fst
  let ⟨b,m⟩ := boddDiv2 n
  cases b <;> rfl
