class SomeOption where
(type : Type 0)

variable [SomeOption]

structure SomeConfiguration where
  options : List SomeOption.type
  otherstuff : String

def someFunction : SomeOption.type → SomeConfiguration → SomeConfiguration
  | op, conf => { conf with options := op::conf.options }
