@[codeActionProvider]
def xyz : CodeActionProvider := fun params snap => do
  let rc : RequestContext ‚Üê read
  let ca : CodeAction := {
    title := "xyz"
  }
  let lazy : RequestM CodeAction := do
        -- this is a requestM
        let doc ‚Üê RequestM.readDoc
        return ca

  return #[{
    eager := ca
    lazy? := some <| (EIO.toIO (fun x => x.message)) <| ReaderT.run lazy rc
  }]
