The dot notation works as expected, and we can also write

mutual
def Foo.toString : Foo → String
  | .somefoo => "foo"
  | .bar b => b.toString

def Bar.toString : Bar → String
  | .somebar => "bar"
  | .foobar f b => f.toString ++ b.toString
end
